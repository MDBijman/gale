symbol ::= keyword | identifier | delimiter | integer.

identifier ::= (letter*) - keyword.
letter ::= 'a' | 'b' | 'c' | 'd' | 'f' | 'g' | 'h' | 'i' | 'j' | 'k' | 'l' | 'm' | 'n' | 'o' | 'p' | 'q' | 'r' | 's' | 't' | 'u' | 'v' | 'w' | 'x' | 'y' | 'z'.

delimiter ::= ';'.
keyword ::= 'int'.

integer ::= number+.
number ::= '1' | '2' | '3' | '4' | '5' | '6' | '7' | '8' | '9' | '0'.
